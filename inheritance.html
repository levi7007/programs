5.Write a C++ program to create a class known as "BankAccount" with methods called deposit() and withdraw(). Create a derived class called SavingsAccount that overrides the withdraw() method to prevent withdrawals if the account balance falls below one hundred.


#include<iostream>
using namespace std;
class BankAccount{
protected:
    double balance;
public:
    BankAccount(double b){balance=b;}
    void deposit(double amt){
        balance+=amt;
        cout<<"Deposited: "<<amt<<" | Balance: "<<balance<<endl;
    }
    virtual void withdraw(double amt){
        balance-=amt;
        cout<<"Withdrawn: "<<amt<<" | Balance: "<<balance<<endl;
    }
};

class SavingsAccount:public BankAccount{
public:
    SavingsAccount(double b):BankAccount(b){}
    void withdraw(double amt){
        if(balance-amt<100){
            cout<<"Cannot withdraw "<<amt<<". Minimum balance 100 required."<<endl;
        }else{
            balance-=amt;
            cout<<"Withdrawn: "<<amt<<" | Balance: "<<balance<<endl;
        }
    }
};
int main(){
    BankAccount b(500);
    SavingsAccount s(500);
    b.deposit(200);
    b.withdraw(300);
    s.deposit(200);
    s.withdraw(550); 
    s.withdraw(400); 
    return 0;
}

Output:

Deposited: 200 | Balance: 700
Withdrawn: 300 | Balance: 400
Deposited: 200 | Balance: 700
Cannot withdraw 550. Minimum balance 100 required.
Withdrawn: 400 | Balance: 300

7.Write a C++ program to create a class known as Person with methods called getFirstName() and getLastName(). Create a derived class called Employee that adds a new method named getEmployeeId() and overrides the getLastName() method to include the employee's job title


#include<iostream>
using namespace std;
class Person{
protected:
    string firstName,lastName;
public:
    Person(string f,string l){
        firstName=f;
        lastName=l;
    }
    string getFirstName(){return firstName;}
    virtual string getLastName(){return lastName;}
};

class Employee:public Person{
    int employeeId;
    string jobTitle;
public:
    Employee(string f,string l,int id,string title):Person(f,l){
        employeeId=id;
        jobTitle=title;
    }
    string getLastName(){
        return lastName+" ("+jobTitle+")";
    }
    int getEmployeeId(){return employeeId;}
};

int main(){
    Person p("John","Doe");
    Employee e("Alice","Smith",101,"Manager");

    cout<<"Person: "<<p.getFirstName()<<" "<<p.getLastName()<<endl;
    cout<<"Employee: "<<e.getFirstName()<<" "<<e.getLastName()<<" | ID: "<<e.getEmployeeId()<<endl;
    return 0;
}

Output:

Person: John Doe
Employee: Alice Smith (Manager) | ID: 101

8.Write a C++ program to create a class called Shape with methods called getPerimeter() and getArea(). Create a derived class called Circle that overrides the getPerimeter() and getArea() methods to calculate the area and perimeter of a circle.


#include<iostream>
using namespace std;
class Shape{
public:
    virtual void getPerimeter(){
        cout<<"Perimeter not defined"<<endl;
    }
    virtual void getArea(){
        cout<<"Area not defined"<<endl;
    }
};

class Circle:public Shape{
    float radius;
public:
    Circle(float r){radius=r;}
    void getPerimeter(){
        cout<<"Perimeter of circle="<<2*3.14*radius<<endl;
    }
    void getArea(){
        cout<<"Area of circle="<<3.14*radius*radius<<endl;
    }
};

int main(){
    Shape s;
    Circle c(5);
    s.getPerimeter();
    s.getArea();
    c.getPerimeter();
    c.getArea();
    return 0;
}

Output:

Perimeter not defined
Area not defined
Perimeter of circle=31.4
Area of circle=78.5

9.Write a C++ program to create a vehicle class hierarchy. The base class should be Vehicle, with derived classes Truck, Car and Motorcycle. Each derived class should have properties such as make, model, year, and fuel type. Implement methods for calculating fuel efficiency, distance traveled, and maximum speed.


#include<iostream>
using namespace std;
class Vehicle{
protected:
    string make,model,fuelType;
    int year;
public:
    Vehicle(string mke,string mdl,int yr,string fuel){
        make=mke;model=mdl;year=yr;fuelType=fuel;
    }
    virtual void fuelEfficiency()=0;
    virtual void distanceTraveled(double hours, double speed)=0;
    virtual void maxSpeed()=0;
};

class Truck:public Vehicle{
    double loadCapacity;
public:
    Truck(string m,string mdl,int y,string fuel,double load):Vehicle(m,mdl,y,fuel){
        loadCapacity=load;
    }
    void fuelEfficiency(){
        cout<<"Truck Fuel Efficiency: 8 km/l"<<endl;
    }
    void distanceTraveled(double hours,double speed){
        cout<<"Truck Distance: "<<hours*speed<<" km"<<endl;
    }
    void maxSpeed(){
        cout<<"Truck Max Speed: 120 km/h"<<endl;
    }
};

class Car:public Vehicle{
public:
    Car(string m,string mdl,int y,string fuel):Vehicle(m,mdl,y,fuel){}
    void fuelEfficiency(){
        cout<<"Car Fuel Efficiency: 15 km/l"<<endl;
    }
    void distanceTraveled(double hours,double speed){
        cout<<"Car Distance: "<<hours*speed<<" km"<<endl;
    }
    void maxSpeed(){
        cout<<"Car Max Speed: 180 km/h"<<endl;
    }
};

class Motorcycle:public Vehicle{
public:
    Motorcycle(string m,string mdl,int y,string fuel):Vehicle(m,mdl,y,fuel){}
    void fuelEfficiency(){
        cout<<"Motorcycle Fuel Efficiency: 40 km/l"<<endl;
    }
    void distanceTraveled(double hours,double speed){
        cout<<"Motorcycle Distance: "<<hours*speed<<" km"<<endl;
    }
    void maxSpeed(){
        cout<<"Motorcycle Max Speed: 150 km/h"<<endl;
    }
};

int main(){
    Truck t("Volvo","FH16",2022,"Diesel",20);
    Car c("Toyota","Corolla",2023,"Petrol");
    Motorcycle m("Yamaha","R15",2021,"Petrol");
    t.fuelEfficiency();
    t.distanceTraveled(5,80);
    t.maxSpeed();
cout<<endl;

    c.fuelEfficiency();
    c.distanceTraveled(3,100);
    c.maxSpeed();
    cout<<endl;

    m.fuelEfficiency();
    m.distanceTraveled(2,120);
    m.maxSpeed();
    return 0;
}

Output:

Truck Fuel Efficiency: 8 km/l
Truck Distance: 400 km
Truck Max Speed: 120 km/h

Car Fuel Efficiency: 15 km/l
Car Distance: 300 km
Car Max Speed: 180 km/h

Motorcycle Fuel Efficiency: 40 km/l
Motorcycle Distance: 240 km
Motorcycle Max Speed: 150 km/h

10.Write a C++ program that creates a class hierarchy for employees of a company. The base class should be Employee, with derived classes Manager, Developer, and Programmer. Each derived class should have properties such as name, address, salary, and job title. Implement methods for calculating bonuses, generating performance reports, and managing projects.


#include<iostream>
using namespace std;
class Employee{
protected:
    string name,address,jobTitle;
    double salary;
public:
    Employee(string n,string ad,double sal,string title){
        name=n; address=ad; salary=sal; jobTitle=title;
    }
    virtual void calculateBonus()=0;
    virtual void performanceReport()=0;
    virtual void manageProjects()=0;
};

class Manager:public Employee{
public:
    Manager(string n,string ad,double sal,string title):Employee(n,ad,sal,title){}
    void calculateBonus(){
        cout<<name<<" (Manager) Bonus: "<<salary*0.2<<endl;
    }
    void performanceReport(){
        cout<<name<<" (Manager) Performance: Excellent"<<endl;
    }
    void manageProjects(){
        cout<<name<<" (Manager) manages multiple projects"<<endl;
    }
};

class Developer:public Employee{
public:
    Developer(string n,string ad,double sal,string title):Employee(n,ad,sal,title){}
    void calculateBonus(){
        cout<<name<<" (Developer) Bonus: "<<salary*0.15<<endl;
    }
    void performanceReport(){
        cout<<name<<" (Developer) Performance: Good"<<endl;
    }
    void manageProjects(){
        cout<<name<<" (Developer) contributes to assigned projects"<<endl;
    }
};

class Programmer:public Employee{
public:
    Programmer(string n,string ad,double sal,string title):Employee(n,ad,sal,title){}
    void calculateBonus(){
        cout<<name<<" (Programmer) Bonus: "<<salary*0.1<<endl;
    }
    void performanceReport(){
        cout<<name<<" (Programmer) Performance: Very Good"<<endl;
    }
    void manageProjects(){
        cout<<name<<" (Programmer) writes and tests code for projects"<<endl;
    }
};

int main(){
    Manager m("Alice","New York",80000,"Manager");
    Developer d("Bob","San Francisco",60000,"Developer");
    Programmer p("Charlie","Los Angeles",50000,"Programmer");
    m.calculateBonus();
    m.performanceReport();
    m.manageProjects();
    cout<<endl;

    d.calculateBonus();
    d.performanceReport();
    d.manageProjects();
    cout<<endl;

    p.calculateBonus();
    p.performanceReport();
    p.manageProjects();

    return 0;
}

Output:

Alice (Manager) Bonus: 16000
Alice (Manager) Performance: Excellent
Alice (Manager) manages multiple projects

Bob (Developer) Bonus: 9000
Bob (Developer) Performance: Good
Bob (Developer) contributes to assigned projects

Charlie (Programmer) Bonus: 5000
Charlie (Programmer) Performance: Very Good
Charlie (Programmer) writes and tests code for projects
