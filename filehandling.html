8.Write a C++ program to sort the lines of a text file in alphabetical order.


#include<iostream>
#include<fstream>
using namespace std;
int main(){
string a[10],t;
int i,j,n=0;
fstream f("file",ios::in);
if(f.is_open()){
    while(getline(f,a[n])){
    n++;
}
}
f.close();
for(i=0;i<n;i++){
    for(j=i+1;j<n;j++){
        if(a[i]>a[j]){ 
            t=a[i];
            a[i]=a[j];
            a[j]=t;
        }
    }
}
fstream f2("file",ios::out);
for(i=0;i<n;i++){
    f2<<a[i]<<endl;
}
cout<<"data sorted sucessfully";
f2.close();
return 0;
}

15.Write a C++ program to calculate the average of numbers stored in a file.


#include<iostream>
#include<fstream>
using namespace std;
int main(){
char ch;
int y=0,count=0,avg=0;
fstream f("file.txt",ios::in);
if(f.is_open()){
    while(!f.eof()){
        ch=f.get();
        if(ch>='0'&&ch<='9'){
            y=y+(ch-'0');
            count++;
        }
    }
avg=y/count;
cout<<avg;
}

    return 0;
}

Output:

4
