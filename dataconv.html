11.C++ Program to convert double to String


#include<iostream>
#include<cmath>
using namespace std;
int main(){
    long long bin;
    cin>>bin;
    int dec=0,i=0,rem;
    while(bin>0){
        rem=bin%10;
        dec+=rem*pow(2,i);
        bin/=10;
        i++;
    }
    cout<<dec<<endl;
    return 0;
}

Output:

Input: 1011
Output: 11

12.C++ Program to convert Decimal to Binary


#include<iostream>
using namespace std;
int main(){
    int dec;
    cin>>dec;
    int bin[32],i=0;
    while(dec>0){
        bin[i]=dec%2;
        dec/=2;
        i++;
    }
    for(int j=i-1;j>=0;j--) cout<<bin[j];
    cout<<endl;
    return 0;
}

Output:

Input: 13
Output: 1101

15.C++ Program to convert Octal to Decimal


#include<iostream>
#include<cmath>
using namespace std;
int main(){
    int oct;
    cin>>oct;
    int dec=0,i=0,rem;
    while(oct>0){
        rem=oct%10;
        dec+=rem*pow(8,i);
        oct/=10;
        i++;
    }
    cout<<dec<<endl;
    return 0;
}

Output:

Input: 17
Output: 15
