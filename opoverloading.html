1.Define a class Complex to represent complex numbers. Overload the + operator to add two complex numbers.


#include<iostream>
using namespace std;
class Complex {
private:
    float real,imag;
public:
    Complex(float r =0,float i=0){
        real=r;
        imag=i;
    }
        Complex operator +(Complex c){
        Complex temp;
        temp.real=real+c.real;
        temp.imag=imag+c.imag;
        return temp;
    }
    void display(){
        cout<<real<<"+"<<imag<<"i"<<endl;
    }
};
int main(){
    Complex c1(2,9),c2(5,5),c3;
    c3=c1+c2;
    cout<<"Sum of complex numbers: ";
    c3.display();
    return 0;
}

Output:

Sum of complex numbers: 7 + 14i

2.Create a class Fraction to represent fractions. Overload the + operator to add two fractions.


#include<iostream>
using namespace std;
class Fraction{
    float num;        
    float den;         
public:
    Fraction(float n=0, float d=1){
        num=n;
        den=d;
    }
    Fraction operator +(Fraction  &obj){
        Fraction temp;
        temp.num = num*obj.den+obj.num*den;
        temp.den = den*obj.den;
        return temp;
    }
    void display(){
        cout<<num<<"/"<<den<<endl;
    }
};
int main(){
    Fraction f1(2,6),f2(7,5),f3;
    f3=f1+f2;
    cout<<"Addtion of Both Fraction numbers are ";
    f3.display();
    return 0;
}

Output:

Addtion of Both Fraction numbers are 52/30

3.Define a class Matrix to represent matrices. Overload the * operator to multiply two matrices.


#include <iostream>
using namespace std;
class Matrix{
    int row, colm;
    int arr[10][10];
public:
    void input(){
        cout<<"Enter rows: ";
        cin>>row;
        cout<<"Enter columns: ";
        cin>>colm;
        for(int i=0;i<row;i++){
            for(int j=0;j<colm;j++){
                cin>>arr[i][j];   
            }
        }
    }
    Matrix operator *(Matrix obj){
        Matrix result;
        if(colm!=obj.row){
            cout<<"Multiplication not possible!";
            return result;
        }
        result.row=row;
        result.colm=obj.colm;
        for(int i=0;i<row;i++){
            for(int j=0;j<obj.colm;j++){
                result.arr[i][j]=0;
                for(int k=0;k<colm;k++){
                    result.arr[i][j]+=arr[i][k]*obj.arr[k][j];
                }
            }
        }
        return result;
    }
    void display(){
        for(int i=0;i<row;i++){
            for(int j=0;j<colm;j++){
                cout<<arr[i][j]<<" ";
            }
            cout<<endl;
        }
    }
};
int main(){
    Matrix obj1,obj2,result;
    cout<<"Enter first matrix:\n";
    obj1.input();
    cout<<"Enter second matrix:\n";
    obj2.input();
    result=obj1*obj2;
    cout<<"Result matrix:\n";
    result.display();
    return 0;
}

    int ruppes;
    int paise;
    public:
    Money(){}
    Money(int rp,int ps){
        ruppes=rp;
        paise=ps;
    }
   Money operator +(Money &obj){
        Money res;
        res.ruppes=ruppes+obj.ruppes;
        res.paise=paise+obj.paise;
        if(res.paise>=100){
            res.ruppes+=res.paise/100;
            res.paise=res.paise%100;
        }

        cout<<"Total Money is : "<<ruppes<<"."<<paise;

};

8.Define a class Time to represent time. Overload the << operator to display the time in HH:MM format.


#include<iostream>
using namespace std;
class Time{
    int hour;
    int minute;
    public:
    Time(int h,int m){
        hour=h;
        minute=m;
    }
    friend ostream& operator<<(ostream& out, Time t) {
        if(t.hour<10)out<<"0";
        out<<t.hour<<":";
        if(t.minute<10)out<<"0";
        out<<t.minute;
        return out;
    }
};
int main(){
    Time t1(12,34);
    cout<<t1;
}

Output:

12:34
